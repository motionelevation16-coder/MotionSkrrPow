Quick Start

From zero to scoring in 60 seconds

Grab your API key from RapidAPI and make your first call.

import requests

url = "https://hookscore.p.rapidapi.com/score"

payload = {
    "hook": "5 habits that made me $10k this month",
    "platform": "tiktok"
}

headers = {
    "x-rapidapi-key": "YOUR_API_KEY_HERE",
    "x-rapidapi-host": "hookscore.p.rapidapi.com",
    "Content-Type": "application/json"
}

response = requests.post(url, json=payload, headers=headers)
data = response.json()

print(f"Score: {data['score']}/100")
print(f"Grade: {data['grade']}")
print(f"Feedback: {data['feedback']}")

# Output:
# Score: 84/100
# Grade: A-
# Feedback: Strong specificity. Consider adding urgency...
curl -X POST https://hookscore.p.rapidapi.com/score \
  -H "x-rapidapi-key: YOUR_API_KEY_HERE" \
  -H "x-rapidapi-host: hookscore.p.rapidapi.com" \
  -H "Content-Type: application/json" \
  -d '{
    "hook": "5 habits that made me $10k this month",
    "platform": "tiktok"
  }'

# Response:
# {
#   "score": 84,
#   "grade": "A-",
#   "platform": "tiktok",
#   "strengths": ["Specific number", "Income claim", "Implied story"],
#   "weaknesses": ["Overused framing", "Missing urgency"],
#   "feedback": "Strong hook. Tweak the income phrasing for originality."
# }
